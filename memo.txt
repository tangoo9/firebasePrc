// db 호출 , get Data
firebase.initializeApp(firebaseConfig);

const db = firebase.firestore()
db.collection('product').get().then((result) => {
    // console.log(result)
    result.forEach((res) => {
        console.log(res.data())
    })
})

//db에 create하기
db.collection('product').doc('상품3').set({ 유통기한: '만료됨' })
db.collection('product').add({ 유통기한: '만료됨' })

add로 쓸경우 doc은 자동으로 이름이 정해짐.
